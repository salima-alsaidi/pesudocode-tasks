CLASS MEDICALCLAIM INHERITS CLAIM
    ATTRIBUTES
        TREATMENTCOST:TEXT
        MEDICALCOVERAGELIMIT:TEXT

    CONSTRUCTOR(TREATMENTCOST, MEDICALCOVERAGELIMIT)
      SUPER(NUMBER, DATE)
      SET THIS.TREATMENTCOST TO TREATMENTCOST
      SET THIS.MEDICALCOVERAGELIMIT TO MEDICALCOVERAGELIMIT
   END CONSTRUCTOR 

    METHODS
    OVERRIDE  calculatePayout(payout)
        IF TREATMENTCOST LESS THAN MEDICALCOVERAGELIMIT THEN 
            RETURN TREATMENTCOST
        ELSE
            RETURN MEDICALCOVERAGELIMIT 
        END IF 
      END METHOD  
END CLASS


CLASS PROPERTYCLAIM INHERITS CLAIM
    ATTRIBUTES
        DAMAGEVALUE:TEXT
        INSPECTIONREPORT:TEXT

    CONSTRUCTOR(DAMAGEVALUE, INSPECTIONREPORT)
     SUPER(NUMBER, DATE)
     SET THIS.DAMAGEVALUE TO DAMAGEVALUE
     SET THIS.INSPECTIONREPORT TO INSPECTIONREPORT
    END CONSTRUCTOR
    METHODS
    OVERRIDE calculatePayout(payout)
        SET REDUCTION TO DAMAGEVALUE * INSPECTIONREPORT.GETREDUCTIONPERCENTAGE()
        RETURN DAMAGEVALUE - REDUCTION
     END METHOD 

END CLASS

CLASS ROLLERCOASTER INHERITS RIDE
    ATTRIBUTES
        CARCAPACITY : NUMBER
        RIDECYCLETIME : NUMBER

    CONSTRUCTOR(NAME, LOCATION, PEOPLEINLINE, LOG, CARCAPACITY, RIDECYCLETIME)
        SUPER(NAME, LOCATION, PEOPLEINLINE, LOG)
        SET THIS.CARCAPACITY TO CARCAPACITY
        SET THIS.RIDECYCLETIME TO RIDECYCLETIME
    END CONSTRUCTOR

    METHODS
        OVERRIDE calculateWaitTime()
            SET WAITTIME TO (PEOPLEINLINE / CARCAPACITY) * RIDECYCLETIME
            RETURN WAITTIME
        END METHOD
END CLASS


CLASS WATERRIDE INHERITS RIDE
    ATTRIBUTES
        BOATCAPACITY : NUMBER
        RIDECYCLETIME : NUMBER
        LOADINGTIME : NUMBER

    CONSTRUCTOR(NAME, LOCATION, PEOPLEINLINE, LOG, BOATCAPACITY, RIDECYCLETIME, LOADINGTIME)
        SUPER(NAME, LOCATION, PEOPLEINLINE, LOG)
        SET THIS.BOATCAPACITY TO BOATCAPACITY
        SET THIS.RIDECYCLETIME TO RIDECYCLETIME
        SET THIS.LOADINGTIME TO LOADINGTIME
    END CONSTRUCTOR

    METHODS
        OVERRIDE calculateWaitTime()
            SET WAITTIME TO (PEOPLEINLINE / BOATCAPACITY) * RIDECYCLETIME
            SET WAITTIME TO WAITTIME + LOADINGTIME
            RETURN WAITTIME
        END METHOD
END CLASS


CLASS VIPACCESSRIDE INHERITS ROLLERCOASTER
    ATTRIBUTES
        VIPPEOPLE : NUMBER
        STANDARDPEOPLE : NUMBER

    CONSTRUCTOR(NAME, LOCATION, PEOPLEINLINE, LOG, CARCAPACITY, RIDECYCLETIME, VIPPEOPLE, STANDARDPEOPLE)
        SUPER(NAME, LOCATION, PEOPLEINLINE, LOG, CARCAPACITY, RIDECYCLETIME)
        SET THIS.VIPPEOPLE TO VIPPEOPLE
        SET THIS.STANDARDPEOPLE TO STANDARDPEOPLE
    END CONSTRUCTOR

    METHODS
        OVERRIDE calculateWaitTime()
            SET VIPWAIT TO (VIPPEOPLE / CARCAPACITY) * RIDECYCLETIME
            SET STANDARDWAIT TO (STANDARDPEOPLE / CARCAPACITY) * RIDECYCLETIME

            DISPLAY "VIP WAIT TIME = ", VIPWAIT
            DISPLAY "STANDARD WAIT TIME = ", STANDARDWAIT
        END METHOD
END CLASS

CLASS HOURLYINVOICE INHERITS INVOICE
    ATTRIBUTES
        HOURSWORKED : NUMBER
        HOURLYRATE : NUMBER
        APPROVED : BOOLEAN

    CONSTRUCTOR(NUMBER, DATE, CLIENT, TERMS, HOURSWORKED, HOURLYRATE, APPROVED)
        SUPER(NUMBER, DATE, CLIENT, TERMS)
        SET THIS.HOURSWORKED TO HOURSWORKED
        SET THIS.HOURLYRATE TO HOURLYRATE
        SET THIS.APPROVED TO APPROVED
    END CONSTRUCTOR

    METHODS
        OVERRIDE calculateTotal()
            IF APPROVED EQUALS TRUE THEN
                RETURN HOURSWORKED * HOURLYRATE
            ELSE
                RETURN 0
            END IF
        END METHOD

CLASS RETAINERINVOICE INHERITS FIXEDPRICEINVOICE
    ATTRIBUTES
        INCLUDEDHOURS : NUMBER
        EXTRAHOURS : NUMBER
        HOURLYRATE : NUMBER

    CONSTRUCTOR(NUMBER, DATE, CLIENT, TERMS, PROJECTFEE, INCLUDEDHOURS, EXTRAHOURS, HOURLYRATE)
        SUPER(NUMBER, DATE, CLIENT, TERMS, PROJECTFEE)
        SET THIS.INCLUDEDHOURS TO INCLUDEDHOURS
        SET THIS.EXTRAHOURS TO EXTRAHOURS
        SET THIS.HOURLYRATE TO HOURLYRATE
    END CONSTRUCTOR

    METHODS
        OVERRIDE calculateTotal()
            IF EXTRAHOURS GREATER THAN INCLUDEDHOURS THEN 
             SET TOTAL TO PROJECTFEE
             SET TOTAL TO TOTAL + (EXTRAHOURS * HOURLYRATE)
             DISPLAY TOTAL
           END IF 
        END METHOD
END CLASS

 CLASS SERVER
    ATTRIBUTES
        SERIALNUMBER : TEXT
        UPTIME : NUMBER
        HARDWARESPECIFICATION:TEXT

    CONSTRUCTOR(SERIALNUMBER, UPTIME, HARDWARE)
        SET THIS.SERIALNUMBER TO SERIALNUMBER
        SET THIS.UPTIME TO UPTIME
        SET HARDWARESPECIFICATION TO HARDWARE
    END CONSTRUCTOR

    METHODS
        calculateLoad()
            RETURN 0
        END METHOD

        SHOWDETAILS()
            DISPLAY SERIALNUMBER
            DISPLAY UPTIME
        END METHOD
END CLASS


CLASS WEBSERVER INHERITS SERVER
    ATTRIBUTES
        ACTIVECONNECTIONS : NUMBER
        MAXCONNECTIONS : NUMBER

    CONSTRUCTOR(SERIALNUMBER, UPTIME, HARDWARE, ACTIVECONNECTIONS, MAXCONNECTIONS)
        SUPER(SERIALNUMBER, UPTIME, HARDWARE)
        SET THIS.ACTIVECONNECTIONS TO ACTIVECONNECTIONS
        SET THIS.MAXCONNECTIONS TO MAXCONNECTIONS
    END CONSTRUCTOR

    METHODS
        OVERRIDE calculateLoad()
            RETURN (ACTIVECONNECTIONS / MAXCONNECTIONS) * 100
        END METHOD
END CLASS


CLASS DATABASESERVER INHERITS SERVER
    ATTRIBUTES
        QUERYTHROUGHPUT : NUMBER
        MAXQUERIESPERSECOND : NUMBER

    CONSTRUCTOR(SERIALNUMBER, UPTIME, HARDWARE, QUERYTHROUGHPUT, MAXQUERIESPERSECOND)
        SUPER(SERIALNUMBER, UPTIME, HARDWARE)
        SET THIS.QUERYTHROUGHPUT TO QUERYTHROUGHPUT
        SET THIS.MAXQUERIESPERSECOND TO MAXQUERIESPERSECOND
    END CONSTRUCTOR

    METHODS
        OVERRIDE calculateLoad()
            IF QUERYTHROUGHPUT GREATER THAN OR EQUAL MAXQUERIESPERSECOND THEN
                DISPLAY "SERVER AT FULL CAPACITY"
                RETURN 100
            ELSE
                RETURN (QUERYTHROUGHPUT / MAXQUERIESPERSECOND) * 100
            END IF
        END METHOD
END CLASS


CLASS CACHESERVER INHERITS DATABASESERVER
    ATTRIBUTES
        CACHEHITRATIO : NUMBER

    CONSTRUCTOR(SERIALNUMBER, UPTIME, HARDWARE, QUERYTHROUGHPUT, MAXQUERIESPERSECOND, CACHEHITRATIO)
        SUPER(SERIALNUMBER, UPTIME, HARDWARE, QUERYTHROUGHPUT, MAXQUERIESPERSECOND)
        SET THIS.CACHEHITRATIO TO CACHEHITRATIO
    END CONSTRUCTOR

    METHODS
        OVERRIDE calculateLoad()
            SET LOAD TO SUPER.calculateLoad()
            SET EFFECTIVELOAD TO LOAD - (LOAD * CACHEHITRATIO / 100)
            RETURN EFFECTIVELOAD
        END METHOD
END CLASS
