Task 288: Bookshop
CLASS BOOK 
 ATTRIBUTE 
   BOOK_ID:NUMBER
   BOOK_TITLE:TEXT
   BOOK_AUTHOR:TEXT
   PRICE:TEXT
 CONSTRUCTOR(ID:NUMBER, T:TEXT, A:TEXT, P:TEXT)
  SET BOOK_ID TO ID 
  SET BOOK_TITLE TO T
  SET BOOK_AUTHOR TO A
  SET PRICE P
 END CONSTRUCTOR
 METHOD GET_PRICE()
        RETURN PRICE
    END METHOD

END CLASS
 
CLASS CUSTOMER 
 ATTRIBUTE 
   CUSTOMER_ID:NUMBER
   CUSTOMER_NAME:TEXT
   PURCHASEHISTORY:LIST <TEXT> []
   BOOK_PRICE:TEXT
 CONSTRUCTOR (ID:NUMBER, N:TEXT, HTEXT, P:TEXT)
  SET CUSTOMER_ID TO ID 
  SET CUSTOMER_NAME TO N
  SET PURCHASEHISTORY TO LIST OF TEXT
  SET BOOK_PRICE TO P
 END CONSTRUCTOR

 METHODS 
   ADD_PURCHASE(SALE)
        ADD SALE TO PURCHASEHISTORY
    END METHOD
END CLASS

CLASS SALE

    ATTRIBUTE
        SALEID : NUMBER
        BOOK : BOOK
        SELLINGPRICE : NUMBER

    CONSTRUCTOR (ID, B)
        SET SALEID TO ID
        SET BOOK TO B
        SET SELLINGPRICE TO B.GET_PRICE()
    END CONSTRUCTOR

    METHOD GET_DETAILS()
        DISPLAY BOOK.TITLE
        DISPLAY SELLINGPRICE
    END METHOD

END CLASS

START
 BEGIN
    CREATE book1 OF BOOK(1, "HARRY POTTER", "J.K. ROWLING", 10, "FANTASY")
    CREATE book2 OF BOOK(2, "ATOMIC HABITS", "JAMES CLEAR", 15, "SELF HELP")

    CREATE customer1 OF CUSTOMER(101, "ALI")

    CREATE sale1 OF SALE(1, book1)
    customer1.ADD_PURCHASE(sale1)

    CREATE sale2 OF SALE(2, book2)
    customer1.ADD_PURCHASE(sale2)

    customer1.VIEW_HISTORY()
 END
STOP

    METHODS

        END METHOD

    END METHODS

CLASS ATTEMPT

    ATTRIBUTE
        ATTEMPTID : NUMBER
        EXAM : TEXT
        STUDENT : TEXT
        ANSWERS : LIST OF TEXT
        SCORE : NUMBER
        TIMEEXPIRED : TEXT

    CONSTRUCTOR(ID:NUMBER, E:TEXT, S:TEXT, PERIOD:TEXT)
        SET ATTEMPTID TO ID
        SET EXAM TO E
        SET STUDENT TO S
        SET ANSWERS TO LIST[]
        SET SCORE TO 0
        SET TIMEEXPIRED TO PERIOD
    END CONSTRUCTOR

    METHODS

        ADD_ANSWER(ANSWER)
            IF TIMEEXPIRED EQUALS "FALSE" THEN 
                ADD ANSWER TO ANSWERS
            ELSE
                DISPLAY "TIME LIMIT EXPIRED"
            END IF
        END METHOD

        CALCULATE_SCORE()
            SET SCORE TO 0

            REPEAT FROM 1 TO SIZE OF ANSWERS
                IF ANSWERS[INDEX] = EXAM.QUESTIONS[INDEX].CORRECTANSWER THEN
                    SET SCORE TO SCORE + EXAM.QUESTIONS[INDEX].GET_POINTS()
                END IF
            END REPEAT

            RETURN SCORE
        END METHOD

    END METHODS

END CLASS


START
 BEGIN
    CREATE exam1 OF EXAM(1, "ICT EXAM", 60)

    CREATE question1 OF QUESTION(1, "2 + 2 ?", "4", 5)
    CREATE question2 OF QUESTION(2, "Capital of Oman?", "MUSCAT", 5)

    exam1.ADD_QUESTION(question1)
    exam1.ADD_QUESTION(question2)

    CREATE student1 OF STUDENT(101, "ALI")

    CREATE attempt1 OF ATTEMPT(1, exam1, student1)

    student1.ADD_ATTEMPT(attempt1)

    attempt1.ADD_ANSWER("4")
    attempt1.ADD_ANSWER("MUSCAT")

    SET score TO attempt1.CALCULATE_SCORE()

    DISPLAY score
 END 
STOP

Task 291: Co-working Space

CLASS MEMBERSHIPPLAN

    ATTRIBUTE
        PLANID : NUMBER
        PLANNAME : TEXT
        MONTHLYQUOTA : NUMBER

    CONSTRUCTOR(ID:NUMBER, NAME:TEXT, QUOTA:TEXT)
        SET PLANID TO ID
        SET PLANNAME TO NAME
        SET MONTHLYQUOTA TO QUOTA
    END CONSTRUCTOR

    METHODS

        GET_QUOTA()
            RETURN MONTHLYQUOTA
        END METHOD

    END METHODS

END CLASS


CLASS MEMBER

    ATTRIBUTE
        MEMBERID : NUMBER
        NAME : TEXT
        PLAN : MEMBERSHIPPLAN
        BOOKINGS : LIST OF BOOKING

    CONSTRUCTOR(ID:NUMBER, N:TEXT, P:TEXT)
        SET MEMBERID TO ID
        SET NAME TO N
        SET PLAN TO P
        SET BOOKINGS TO  LIST[]
    END CONSTRUCTOR

    METHODS

        ADD_BOOKING(B)
            ADD B TO BOOKINGS
        END METHOD

        GET_REMAINING_HOURS()

            SET USED TO 0

            REPEAT FROM 1 TO SIZE OF BOOKINGS
                SET USED TO USED + BOOKINGS[INDEX].DURATION
            END REPEAT

            RETURN PLAN.GET_QUOTA() - USED

        END METHOD

    END METHODS

END CLASS


CLASS ROOM

    ATTRIBUTE
        ROOMID : NUMBER
        ROOMNAME : TEXT
        BOOKINGS : LIST OF BOOKING

    CONSTRUCTOR(ID, NAME)
        SET ROOMID TO ID
        SET ROOMNAME TO NAME
        SET BOOKINGS TO  LIST[]
    END CONSTRUCTOR

    METHODS

        ADD_BOOKING(B)
            ADD B TO BOOKINGS
        END METHOD

    END METHODS

END CLASS


CLASS BOOKING

    ATTRIBUTE
        BOOKINGID : NUMBER
        MEMBER : MEMBER
        ROOM : TEXT
        DATE : TEXT
        STARTHOUR : NUMBER
        ENDHOUR : NUMBER
        DURATION : NUMBER

    CONSTRUCTOR(ID:NUMBER, M:MEMBER, R:TEXT, D:TEXT, START:NUMBER, D:NUMBER)
        SET BOOKINGID TO ID
        SET MEMBER TO M
        SET ROOM TO R
        SET DATE TO D
        SET STARTHOUR TO START
        SET ENDHOUR TO END
        SET DURATION TO D
    END CONSTRUCTOR

    METHODS

        MAKE_BOOKING()

            IF DURATION LESS THAN OR EQUAL MEMBER.GET_REMAINING_HOURS() THEN

                ADD SELF TO MEMBER.BOOKINGS
                ADD SELF TO ROOM.BOOKINGS

                DISPLAY "BOOKING CONFIRMED"

            ELSE

                DISPLAY "MONTHLY QUOTA EXCEEDED"

            END IF

        END METHOD

    END METHODS

END CLASS

    CREATE room1 OF ROOM("201", "MEETING ROOM A")
