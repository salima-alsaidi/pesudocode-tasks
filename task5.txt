Task 302: Smart Home Devices

CLASS STORAGEUNIT

    ATTRIBUTES
        TOTALSPACE : NUMBER
        USEDSPACE : NUMBER

    CONSTRUCTOR(TOTALSPACE:NUMBER, USEDSPACE:NUMBER)

        SET THIS.TOTALSPACE TO TOTALSPACE
        SET THIS.USEDSPACE TO USEDSPACE

    END CONSTRUCTOR

    METHODS
      RECORDFOOTAGE(SIZE)

        IF USEDSPACE + SIZE LESS THAN OR EQUAL TO TOTALSPACE THEN
            SET USEDSPACE TO USEDSPACE + SIZE
            DISPLAY "RECORDING STARTED."
        ELSE
            DISPLAY "STORAGE FULL. CANNOT RECORD."
        END IF

    END METHOD

END CLASS
