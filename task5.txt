CLASS TRIANGLE INHERITS SHAPE

    ATTRIBUTE
        BASE : NUMBER
        HEIGHT : NUMBER
        SIDE1 : NUMBER
        SIDE2 : NUMBER
        SIDE3 : NUMBER


    CONSTRUCTOR(C : TEXT, P : TEXT, B : NUMBER, H : NUMBER, S1 : NUMBER, S2 : NUMBER, S3 : NUMBER)

        SUPER(C, P)

        SET THIS.BASE TO B
        SET THIS.HEIGHT TO H
        SET THIS.SIDE1 TO S1
        SET THIS.SIDE2 TO S2
        SET THIS.SIDE3 TO S3

    END CONSTRUCTOR


    METHODS

        CALCULATE_AREA()

            SET SEMIPERIMETER TO (SIDE1 + SIDE2 + SIDE3) / 2

            SET AREA TO SQRT(
                SEMIPERIMETER *
                (SEMIPERIMETER - SIDE1) *
                (SEMIPERIMETER - SIDE2) *
                (SEMIPERIMETER - SIDE3)
            )

        END METHOD


        CALCULATE_PERIMETER()

            SET PERIMETER TO SIDE1 + SIDE2 + SIDE3

        END METHOD

END CLASS



CLASS RECTANGLE INHERITS SHAPE

    ATTRIBUTE
        BASE : NUMBER
        HEIGHT : NUMBER


    CONSTRUCTOR(C : TEXT, P : TEXT, B : NUMBER, H : NUMBER)

        SUPER(C, P)

        SET THIS.HEIGHT TO H

    METHODS

        CALCULATE_AREA()

            SET AREA TO BASE * HEIGHT

        END METHOD


        CALCULATE_PERIMETER()

            SET PERIMETER TO (2 * BASE) + (2 * HEIGHT)

        END METHOD

END CLASS

CLASS DEBITCARD INHERITS CARD

    ATTRIBUTES
        ACCOUNT : CUSTOMERACCOUNT

    CONSTRUCTOR(NUM, ED, S, CHIP, ACCOUNT)
        SUPER(NUM, ED, S, CHIP)
        SET THIS.ACCOUNT TO ACCOUNT
    END CONSTRUCTOR

    METHODS

        AUTHORIZEPURCHASE(AMOUNT)
            IF SUPER.AUTHORIZEPURCHASE(AMOUNT) AND ACCOUNT.CHECKBALANCE() >= AMOUNT THEN
                ACCOUNT.DEDUCTAMOUNT(AMOUNT)
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

END CLASS


CLASS CREDITCARD INHERITS CARD

    ATTRIBUTES
        CREDITLIMIT : NUMBER
        PURCHASES : LIST<PURCHASE>

    CONSTRUCTOR(NUM, ED, S, CHIP, LIMIT)
        SUPER(NUM, ED, S, CHIP)
        SET CREDITLIMIT TO LIMIT
        SET PURCHASES TO EMPTY LIST
    END CONSTRUCTOR

    METHODS

        AUTHORIZEPURCHASE(AMOUNT)

            IF SUPER.AUTHORIZEPURCHASE(AMOUNT) AND CALCULATEOUTSTANDINGBALANCE() + AMOUNT <= CREDITLIMIT THEN

                CREATE OBJECT P OF PURCHASE(AMOUNT, CURRENTDATE)

                ADD P TO PURCHASES

                RETURN TRUE

            ELSE

                RETURN FALSE

            END IF

        END METHOD

        CALCULATEOUTSTANDINGBALANCE()

            SET TOTAL TO 0

            REPEAT FROM I = 0 TO SIZE OF PURCHASES - 1
                SET TOTAL TO TOTAL + PURCHASES[I].GETAMOUNT()
            END REPEAT

            RETURN TOTAL

        END METHOD

END CLASS
