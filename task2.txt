 Task 281: School Classroom 
 CLASS CLASSES 
  ATTRIBUTE 
   ID:NUMBER 
   SUBJECT:TEXT
   TEACHER:TEXT
   STUDENT:LIST<>[]
   CLASS_PERIOD:TEXT
   CLASS_CAPACITY:NUMBER
  METHODS 
   GET_STUDENT()  
      DISPLAY STUDENT
 END METHOD
  METHODS 
   RTEMOVE_STUDENT()
    IF  STUDENT CONTAINS STUDENT THEN 
      CREATE OBJECT NEWSTUDENT OF CLASSES 
      REMOVE NEWSTUDENT FROM STUDENT 
    ELSE 
      DISPLAY "STUDENT NOT FOUND"
    END IF  
  END METHOD 
  
 CLASS STUDENTS
  ATTRIBUTE 
   ID:NUMBER
   NAME:TEXT
   CLASSESS:LIST<TEXT>[]
   GRADES:LIST<TEXT>[]
  METHODS 
   ADDCLASS()
     CREATE OBJECT NEWCLASS OF STUDENTS
     ADD NEWCLASS TO CLASSESS
     DISPLAY CLASSESS
   END METHOD 
  METHODS 
   REMOVECLASS()
     CREATE OBJECT NEWCLASS OF STUDENTS
     REMOVE NEWCLASS FROM CLASSESS
     DISPLAY CLASSESS
   END METHOD
     
   METHODS 
   ADDGRADS()
     CREATE OBJECT GRADE OF STUDENTS
     ADD GRADE TO GRADES
     DISPLAY CLASSESS
   END METHOD

  CLASS GRADE 
    ATTRIBUTE 
     NAME:TEXT 
     SCORE:NUMBER 
     WEIGHT:TEXT
   METHODS 
    GETWIGHTSCORE()
       DISPLAY SCORE+WIGHT
   END METHOD 

 CLASS TEACHER 
  ATTRIBUTE 
   PRIVATE ID:NUMBER 
   PRIVATE NAME:TEXT
   PRIVATE SUBJECT"TEXT
   PRIVATE CLASSES:TEXT
  METHODS 
   ASSSIGNCLASS() 
    CREATE OBJECT NEWCLASS TO TEACHER 
    ADD NEWCLASS TO CLASSES
  END METHODS 
  METHODS 
   SHOWTEACHER() 
    DISPLAY "ID"+ID
    DISPLAY "NAME"+NAME
    DISPLAY "SUBJECT"+SUBJECT
    DISPLAY "CLASSES"+CLASSES
  END METHODS 

  CLASS SUBJECT
   ATTRIBUTE 
    SUBJECT_NAME:TEXT
    SUBJECT_CODE:TEXT
    SYLLABUS"TEXT"
   METHODS
    SHOWSUBJECT() 
    DISPLAY SUBJECT 
   END METHOD 

START 
  BEGIN 
   CREATE OBJECT teacher1 OF TEACHER(011, "ALI", "MATH", “CLASS 59”)
   CREATE OBJECT class1 OF CLASSES(“C011”, "ALI", "MATH", “teacher1”)
   CREATE OBJECT student1 OF STUDENTS(27, "ALI", "class1", “89”)
   CREATE OBJECT grade1 OF GRADE(“QUIZ1”, 86,44)
   teacher1.ASSSIGNCLASS()
   class1.ADDCLASS()
   student1.SHOWSUBJECT() 
   grade1.GETWIGHTSCORE()
   
  END
STOP
