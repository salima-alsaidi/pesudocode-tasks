CLASS SPRINT
    ATTRIBUTES
        SPRINTID:NUMBER
        TASKS:LIST

    CONSTRUCTOR(ID:NUMBER)
        SET SPRINTID TO ID
        SET TASKS TO EMPTY LIST
    END CONSTRUCTOR

    METHOD ADDTASK(TASK)
        ADD TASK TO TASKS
    END METHOD

    METHOD CALCULATEPROGRESS()
        SET DONE TO 0

        FOR EACH TASK IN TASKS DO
            IF TASK.STATUS EQUALS "COMPLETED" THEN
                SET DONE TO DONE + 1
            END IF
        END FOR

        IF SIZE OF TASKS > 0 THEN
            RETURN (DONE / SIZE OF TASKS) * 100
        ELSE
            RETURN 0
        END IF
    END METHOD
END CLASS

        SET ACTIVITIES TO EMPTY LIST
    END CONSTRUCTOR

    METHOD LOGACTIVITY(ACTIVITY)
        ADD ACTIVITY TO ACTIVITIES
    END METHOD

    METHOD CALCULATEYIELD()
        SET SCORE TO 0
        SET TOTAL TO SIZE OF ACTIVITIES

        FOR EACH ACTIVITY IN ACTIVITIES DO
            IF ACTIVITY.TYPE EQUALS "WATERING" THEN
                SET SCORE TO SCORE + 1
            END IF
        END FOR

        IF TOTAL LESS THAN 0 THEN
            DISPLAY (SCORE / TOTAL) * CROP.EXPECTEDYIELD
        ELSE
            DISPLAY 0
        END IF
    END METHOD
END CLASS
