CLASS TASK

    ATTRIBUTE
        TASKID : NUMBER
        TITLE : TEXT
        STATUS : TEXT
        DEVELOPER : DEVELOPER

    CONSTRUCTOR (ID, T, DEV)
        SET TASKID TO ID
        SET TITLE TO T
        SET DEVELOPER TO DEV
        SET STATUS TO "TO DO"
    END CONSTRUCTOR

    METHODS

        UPDATE_STATUS(NEWSTATUS)
            IF STATUS = "TO DO" AND NEWSTATUS = "IN PROGRESS" THEN
                SET STATUS TO NEWSTATUS
            ELSE IF STATUS = "IN PROGRESS" AND NEWSTATUS = "DONE" THEN
                SET STATUS TO NEWSTATUS
            ELSE IF STATUS = "DONE" AND NEWSTATUS = "COMPLETED" THEN
                SET STATUS TO NEWSTATUS
            ELSE
                DISPLAY "INVALID STATUS TRANSITION"
            END IF
        END METHOD

        MARK_COMPLETE()
            IF STATUS = "DONE" THEN
                SET STATUS TO "COMPLETED"
            ELSE
                DISPLAY "TASK NOT READY"
            END IF
        END METHOD

    END METHODS

END CLASS

CLASS SPRINT

    ATTRIBUTE
        SPRINTID : NUMBER
        TASKS : LIST OF TASK

    CONSTRUCTOR (ID)
        SET SPRINTID TO ID
        SET TASKS TO EMPTY LIST
    END CONSTRUCTOR

    METHODS

        ADD_TASK(TASK)
            ADD TASK TO TASKS
        END METHOD

        CALCULATE_PROGRESS()
            SET DONE TO 0

            REPEAT FROM 1 TO SIZE OF TASKS
                IF TASKS[INDEX].STATUS = "COMPLETED" THEN
                    SET DONE TO DONE + 1
                END IF
            END REPEAT

            IF SIZE OF TASKS > 0 THEN
                RETURN (DONE / SIZE OF TASKS) * 100
            ELSE
                RETURN 0
            END IF
        END METHOD

    END METHODS

END CLASS

CLASS WORKER

    ATTRIBUTE
        WORKERID : NUMBER
        NAME : TEXT
        DAILYLIMIT : NUMBER
        ASSIGNEDACTIVITIES : LIST OF ACTIVITY

    CONSTRUCTOR (ID, N, LIMIT)
        SET WORKERID TO ID
        SET NAME TO N
        SET DAILYLIMIT TO LIMIT
        SET ASSIGNEDACTIVITIES TO EMPTY LIST
    END CONSTRUCTOR

    METHODS

        CAN_ASSIGN()
            IF SIZE OF ASSIGNEDACTIVITIES < DAILYLIMIT THEN
                RETURN TRUE
            ELSE
                RETURN FALSE
            END IF
        END METHOD

        ASSIGN_ACTIVITY(ACTIVITY)
            IF SIZE OF ASSIGNEDACTIVITIES < DAILYLIMIT THEN
                ADD ACTIVITY TO ASSIGNEDACTIVITIES
            ELSE
                DISPLAY "DAILY LIMIT EXCEEDED"
            END IF
        END METHOD

    END METHODS

END CLASS


CLASS CROP

    ATTRIBUTE
        CROPID : NUMBER
        NAME : TEXT
        WATERNEED : NUMBER
        EXPECTEDYIELD : NUMBER

    CONSTRUCTOR (ID, N, WATER, YIELD)
        SET CROPID TO ID
        SET NAME TO N
        SET WATERNEED TO WATER
        SET EXPECTEDYIELD TO YIELD
    END CONSTRUCTOR

END CLASS


CLASS ACTIVITY

    ATTRIBUTE
        ACTIVITYID : NUMBER
        TYPE : TEXT
        DATE : TEXT
        WORKER : WORKER
        CROP : CROP
        WATERAMOUNT : NUMBER

    CONSTRUCTOR (ID, T, D, W, C, WATER)
        SET ACTIVITYID TO ID
        SET TYPE TO T
        SET DATE TO D
        SET WORKER TO W
        SET CROP TO C
        SET WATERAMOUNT TO WATER
    END CONSTRUCTOR

    METHODS

        VALIDATE_WATER()
            IF WATERAMOUNT <= CROP.WATERNEED THEN
                DISPLAY "WATERING OK"
            ELSE
                DISPLAY "INSUFFICIENT WATER"
            END IF
        END METHOD

    END METHODS

END CLASS


CLASS FIELD

    ATTRIBUTE
        FIELDID : NUMBER
        CROP : CROP
        ACTIVITIES : LIST OF ACTIVITY

    CONSTRUCTOR (ID, C)
        SET FIELDID TO ID
        SET CROP TO C
        SET ACTIVITIES TO EMPTY LIST
    END CONSTRUCTOR

    METHODS

        LOG_ACTIVITY(ACTIVITY)
            ADD ACTIVITY TO ACTIVITIES
        END METHOD

        CALCULATE_YIELD()
            SET SCORE TO 0
            SET TOTAL TO SIZE OF ACTIVITIES

            IF TOTAL > 0 THEN

                REPEAT FROM 1 TO TOTAL
                    IF ACTIVITIES[INDEX].TYPE = "WATERING" THEN
                        SET SCORE TO SCORE + 1
                    END IF
                END REPEAT

                DISPLAY (SCORE / TOTAL) * CROP.EXPECTEDYIELD

            ELSE
                DISPLAY 0
            END IF
        END METHOD

    END METHODS

END CLASS
