Task 303: Musical Instruments Store 
CLASS Instrument
    ATTRIBUTE
        instrumentName : TEXT
        brand : TEXT
        price : NUMBER

    CONSTRUCTOR(name : TEXT, brand : TEXT, price : NUMBER)
        SET instrumentName TO name
        SET THIS.brand TO brand
        SET THIS.price TO price
    END CONSTRUCTOR

    METHODS
        showDescription()
        END METHOD

        showDetails()
            DISPLAY "Name: " + instrumentName
            DISPLAY "Brand: " + brand
            DISPLAY "Price: " + price
        END METHOD
END CLASS

CLASS MusicStore
    ATTRIBUTE
        storeID : TEXT
        instruments : LIST<Instrument>

    CONSTRUCTOR(storeID : TEXT)
        SET THIS.storeID TO storeID
        SET instruments TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        addInstrument(instrument : Instrument)
            ADD instrument TO instruments
        END METHOD

        removeInstrument(instrument : Instrument)
            REMOVE instrument FROM instruments
        END METHOD

        displayInstruments()
            FOR EACH instrument IN instruments
                instrument.showDetails()
                instrument.showDescription()
            END FOR
        END METHOD
END CLASS


START
BEGIN
    CREATE OBJECT store OF MusicStore("MS001")

    CREATE OBJECT guitar OF Guitar("Guitar", "Yamaha", 350.0, 6)
    CREATE OBJECT drum OF Drum("Drum", "Pearl", 1200.0, 5)
    CREATE OBJECT keyboard OF Keyboard("Keyboard", "Casio", 500.0, 61)

    store.addInstrument(guitar)
    store.addInstrument(drum)
    store.addInstrument(keyboard)

    store.displayInstruments()
END
STOP

        SET THIS.USERNAME TO USERNAME

    METHODS

        ADDWORKOUT(WORKOUT)
            ADD WORKOUT TO WORKOUTS
        END METHOD

        REMOVEWORKOUT(WORKOUT)
            REMOVE WORKOUT FROM WORKOUTS
        END METHOD

            REPEAT FROM I = 1 TO SIZE OF WORKOUTS

            END REPEAT

CLASS WATERBILL INHERITS BILL

    ATTRIBUTES
        LITERSCONSUMED:TEXT
        FLATRATE:TEXT
        STANDINGCHARGE:TEXT

    CONSTRUCTOR(LITERSCONSUMED, FLATRATE, STANDINGCHARGE)
     SUPER(BILLINGPERIOD, DUEDATE, METERREADING)
     SET THIS.LITERSCONSUMED TO LITERSCONSUMED
     SET THIS.FLATRATE TO FLATRATE
     SET THIS.STANDINGCHARGE TO STANDINGCHARGE
    END CONSTRUCTOR

    METHODS 
     CALCULATEAMOUNT()

      RETURN (LITERSCONSUMED * FLATRATE) + STANDINGCHARGE
     END METHOD 
   DISPLAYBILL()

    DISPLAY "WATER BILL"
    DISPLAY CALCULATEAMOUNT()
   END METHOD 
END CLASS


CLASS GASBILL INHERITS BILL

    ATTRIBUTES
        UNITSCONSUMED:TEXT
        RATE:TEXT
        SEASON:TEXT
        WINTERSURCHARGE:TEXT

    CONSTRUCTOR( UNITSCONSUMED,RATE, SEASON, WINTERSURCHARGE)
      SUPER(BILLINGPERIOD, DUEDATE, METERREADING)
      SET THIS.UNITSCONSUMED TO UNITSCONSUMED
      SET THIS.RATE TO RATE
      SET THIS.SEASON TO SEASON
      SET THIS.WINTERSURCHARGE TO WINTERSURCHARGE
    END CONSTRUCTOR


   METHODS 
   CALCULATEAMOUNT()

   SET TOTAL TO UNITSCONSUMED * RATE

    IF SEASON EQUALS "WINTER" THEN
       SET TOTAL TO TOTAL + WINTERSURCHARGE
    END IF
   
    RETURN TOTAL
  END METHOD

 DISPLAYBILL()

    DISPLAY "GAS BILL"
    DISPLAY CALCULATEAMOUNT()
 END METHOD

END CLASS

CLASS RIDE

    ATTRIBUTES
        PICKUPLOCATION:TEXT
        DROPOFFLOCATION:TEXT
        ROUTE : ROUTE

    CONSTRUCTOR(PICKUPLOCATION, DROPOFFLOCATION, ROUTE)
      SET THIS.PICKUPLOCATION TO PICKUPLOCATION
      SET THIS.DROPOFFLOCATION TO DROPOFFLOCATION
      SET THIS.ROUTE TO ROUTE
    END CONSTRUCTOR

    METHOD CALCULATEFARE()
        RETURN 0

    METHOD DISPLAYRIDE()
        DISPLAY PICKUPLOCATION
        DISPLAY DROPOFFLOCATION

END CLASS


CLASS STANDARDRIDE INHERITS RIDE

    ATTRIBUTES
        DISTANCE:TEXT
        RATEPERKM:TEXT

    CONSTRUCTOR(PICKUPLOCATION, DROPOFFLOCATION,
                ROUTE, DISTANCE, RATEPERKM)

    OVERRIDE METHOD CALCULATEFARE()
        RETURN DISTANCE * RATEPERKM

END CLASS


CLASS PREMIUMRIDE INHERITS RIDE

    ATTRIBUTES
        DISTANCE:TEXT
        PREMIUMRATE:TEXT
        COMFORTSURCHARGE:TEXT

    CONSTRUCTOR(DISTANCE, PREMIUMRATE, COMFORTSURCHARGE)
      SUPER(PICKUPLOCATION, DROPOFFLOCATION, ROUTE)
      SET THIS.DISTANCE TO DISTANCE
      SET THIS.PREMIUMRATE TO PREMIUMRATE
      SET THIS.COMFORTSURCHARGE TO COMFORTSURCHARGE
    END CONSTRUCTOR
     METHODS
    CALCULATEFARE()
        RETURN (DISTANCE * PREMIUMRATE) + COMFORTSURCHARGE
    END METHOD
END CLASS


CLASS SHAREDRIDE INHERITS STANDARDRIDE

    ATTRIBUTES
        PASSENGERCOUNT:TEXT

    CONSTRUCTOR(PASSENGERCOUNT)
      SUPER(DISTANCE, PREMIUMRATE, COMFORTSURCHARGE)
      SUPER(PICKUPLOCATION, DROPOFFLOCATION, ROUTE)
      SET THIS.PASSENGERCOUNT TO PASSENGERCOUNT

    

     METHODS
     CALCULATEFARE()

      SET  TOTALFARE TO DISTANCE * RATEPERKM

        RETURN TOTALFARE / PASSENGERCOUNT
    END METHOD 
END CLASS
